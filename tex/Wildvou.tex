\dr{Wilcoxonův dvouvýběrový test}
\vspace*{2mm}
\newline $X_1,\dots,X_m$ nezávislé, stejné rozdělení, $Y_1,\dots,Y_n$ nezávislé, stejné rozdělení, výběry nezávislé
\newline $H_0:$ $X_k$ a $Y_k$ má stejné rozdělení
\newline
postup:
\newline - hodnotám $X_1,\dots, X_m, Y_1,\dots,Y_n$ přiřadíme pořadí
\newline - $T_1=$ součet pořadí $X_k$, $T_2=$ součet pořadí  $Y_k$,
\newline - $ U_1=mn+\frac12m(m+1) -T_1 \quad  U_2=mn+\frac12n(n+1) -T_2 $
\newline $H_0$  zamítneme $\quad \Longleftrightarrow \quad \min(U_1,U_2) \geq k$
