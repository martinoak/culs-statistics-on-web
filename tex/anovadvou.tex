\dr{dvojné třídění bez interakcí (ANOVA)}
\vspace*{2mm}
\hspace*{+3mm}
\begin{multline*}\begin{matrix}
X_{1,1,1},\dots,X_{1,1,p}\sim N(\mu_{1,1},\sigma^2) \text{ nez.} & X_{1,2,1},\dots,X_{1,2,p}\sim N(\mu_{1,2},\sigma^2) \text{ nez.} &\dots & X_{1,J,1},\dots,X_{1,J,p}\sim N(\mu_{1,J},\sigma^2) \text{ nez.} \\
X_{2,1,1},\dots,X_{2,1,p}\sim N(\mu_{2,1},\sigma^2) \text{ nez.} & X_{2,2,1},\dots,X_{2,2,p}\sim N(\mu_{2,2},\sigma^2) \text{ nez.} &\dots & X_{2,J,1},\dots,X_{2,J,p}\sim N(\mu_{2,J},\sigma^2) \text{ nez.} \\
\hdotsfor{4} \\
X_{I,1,1},\dots,X_{I,1,p}\sim N(\mu_{I,1},\sigma^2) \text{ nez.} & X_{I,2,1},\dots,X_{I,2,p}\sim N(\mu_{I,2},\sigma^2) \text{ nez.} &\dots & X_{I,J,1},\dots,X_{I,J,p}\sim N(\mu_{I,J},\sigma^2) \text{ nez.}
\end{matrix}\end{multline*}
\hspace*{60mm} výběry nezávislé, $I\cdot J\cdot p =n$
\newline (průměry)
\begin{multline*}\begin{matrix}
\bar{X}_{1,1}=\frac1{p}\sum_{k} X_{1,1,k} & \bar{X}_{1,2}=\frac1{p}\sum_{k} X_{1,2,k} & \dots & \bar{X}_{1,J}=\frac1{p}\sum_{k} X_{1,J,k}& \qquad \bar{X}^{A}_{1}=\frac1{p\cdot J}\sum_{kl} X_{1,k,l} \\
\bar{X}_{2,1}=\frac1{p}\sum_{k} X_{2,1,k} & \bar{X}_{2,2}=\frac1{p}\sum_{k} X_{2,2,k} & \dots & \bar{X}_{2,J}=\frac1{p}\sum_{k} X_{2,J,k}& \qquad \bar{X}^{A}_{2}=\frac1{p\cdot J}\sum_{kl} X_{2,k,l} \\
\hdotsfor{5} \\
\bar{X}_{I,1}=\frac1{p}\sum_{k} X_{I,1,k} & \bar{X}_{I,2}=\frac1{p}\sum_{k} X_{I,2,k} & \dots & \bar{X}_{I,J}=\frac1{p}\sum_{k} X_{I,J,k}& \qquad \bar{X}^{A}_{I}=\frac1{p\cdot J}\sum_{kl} X_{I,k,l} \\
\bar{X}^{B}_{1}=\frac1{p\cdot I}\sum_{k,l} X_{k,1,l} & \bar{X}^{B}_{2}=\frac1{p\cdot I}\sum_{k,l} X_{k,2,l} & \dots & \bar{X}^{B}_{J}=\frac1{p\cdot I}\sum_{k,l} X_{k,J,l} & \qquad \bar{X}=\frac1{n}\sum_{klm} X_{k,l,m} \\
\end{matrix}\end{multline*}
\newline \hspace*{5mm} $\mu_{k,l}=a+\alpha_{k}+\beta_{l}\quad \sum_{k}\alpha_{k}=0 \quad \sum_{l}\beta_{l}=0$
\par $H_0$:   $\alpha_1=\alpha_2=\dots=\alpha_{I}=0$
\par $H_0$:   $\beta_1=\beta_2=\dots=\beta_{I}=0$
\vspace*{3mm}\newline
(součty čtverců)
$$\begin{array}{ll}
S_{T}=\sum_{klm} (X_{k,l,m}-\bar{X})^2\quad & f_{T}=n-1 \\
S_{A}=\sum_{k} (\bar{X}^{A}_{k}-\bar{X})^2\cdot J\cdot p\quad & f_{A}=I-1 \\
S_{B}=\sum_{k} (\bar{X}^{B}_{k}-\bar{X})^2\cdot I\cdot p\quad & f_{A}=J-1 \\
S_{e}=S_{T}-S_{A}-S_{B}\quad & f_{e}=n-I-J+1
\end{array}   \qquad F_{A}=\frac{\frac{S_{A}}{f_{A}}}{\frac{S_{e}}{f_{e}}}\qquad F_{B}=\frac{\frac{S_{B}}{f_{B}}}{\frac{S_{e}}{f_{e}}}\qquad
s^2=\frac{S_{e}}{f_{e}}$$

\par $H_0$:  $\alpha_1=\dots = \alpha_I = 0$ zamítneme $\quad\Longleftrightarrow \quad $ $F_{A}\geq F_{f_{A},f_{e}}(1-\alpha)$
\par $H_0$:  $\beta_1=\dots = \beta_I = 0$ zamítneme $\quad\Longleftrightarrow \quad $ $F_{B}\geq F_{f_{B},f_{e}}(1-\alpha)$

\par\noindent\hspace*{5mm}
(Sheffého metoda) třídy {\it k}, {\it l} se liší $\quad\Longleftrightarrow \quad $ $\displaystyle{ | \bar{X}^{A}_{k}- \bar{X}^{A}_{l}|>
\sqrt{\frac{2(I-1)}{JP} s^2  F_{f_{A},f_{e}}(1-\alpha)}}$
\par\noindent\hspace*{5mm}
\phantom{(Sheffého metoda)} třídy {\it k}, {\it l} se liší $\quad\Longleftrightarrow \quad $ $\displaystyle{ | \bar{X}^{B}_{k}- \bar{X}^{B}_{l}|>
\sqrt{\frac{2(J-1)}{IP} s^2  F_{f_{B},f_{e}}(1-\alpha)}}$
