\dr{Kruskal - Wallis test} 
\begin{multline*}\begin{matrix}
X_{1,1},\dots,X_{1,n_1} \text{ independent with the same distribution } \\
X_{2,1},\dots,X_{2,n_2} \text{  independent with the same distribution  } \\
\hdotsfor{1}  \\
X_{I,1},\dots,X_{I,n_{I}} \text{  independent with the same distribution  }
\end{matrix}  \end{multline*}
\hspace*{80mm} independent samples, $n_1+\dots+n_{I}=n$
\newline $H_0$:  all distributions are the same
\newline
process:
\newline - we associate the corresponding order with the values $X_{11},\dots, X_{1n_1}, \dots, X_{I1},\dots, X_{In_I} $ 
\newline - $T_1=$ sum of orders of $X_{1k}$, $T_2=$ sum of orders of  $X_{2k}$,...,$T_I=$ sum of orders of  $X_{Ik}$
\newline - $\displaystyle{ Q=\frac{12}{n(n+1)} \sum_k \frac{T^2_k}{n_k} -3(n+1)\approx \chi^2_{I-1}}$
\newline $H_0$  rejected $\quad \Longleftrightarrow\quad |Q| \geq k$
\newline \phantom{$H_0$  rejected $\quad \Longleftrightarrow$} $\quad |Q| \geq \chi^2_{I-1}(1-\alpha)$
