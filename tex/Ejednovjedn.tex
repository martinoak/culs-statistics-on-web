\dr{one-sample t-test one-tailed}
\vspace*{2mm}
\par $X_{1},\dots,X_{n}\sim N(\mu,\sigma^2)$ independent
$$\bar{X}=\frac1{n}\left(X_1 + \dots + X_{n}\right) \qquad S_{X}^2=\frac1{n-1}\left((X_1-\bar{X})^2 + \dots + (X_{n}-\bar{X})^2\right)$$
$\sum_{k}(X_{k}-\bar{X})^2=   \sum_{k}X_{k}^2- n\cdot \bar{X}^2    $
$$ \qquad T=\frac{\bar{X}-\mu}{\sqrt{\frac{S_{X}^2}{n}}}\sim t_{n-1}\qquad     $$
\par\noindent - one-tailed test for $\mu$ on level $\alpha$
\vspace*{3mm}
\par $H_0$:  $\mu\leq a$ rejected $\quad\Longleftrightarrow\quad$  $T \geq t_{n-1}(1-\alpha)$
\vspace*{2mm}
\par $H_0$:  $\mu\geq a$ rejected $\quad\Longleftrightarrow\quad$  $T \leq t_{n-1}(\alpha) = - t_{n-1}(1-\alpha)$
\vspace*{5mm}
\par\noindent - one-tailed 95\% confident interval for $\mu$


$$\left(\bar{X}-\frac{S_{X}}{\sqrt{n}}\cdot t_{n-1}(0,95);\infty \right)$$
$$\left(-\infty ;\bar{X}+\frac{S_{X}}{\sqrt{n}}\cdot t_{n-1}(0,95) \right)$$
