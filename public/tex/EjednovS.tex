\dr{one-sample (Pearson) test}
\vspace*{2mm}
\par $X_{1},\dots,X_{n}\sim N(\mu,\sigma^2)$ independent
$$\bar{X}=\frac1{n}\left(X_1 + \dots + X_{n}\right) \qquad S_{X}^2=\frac1{n-1}\left((X_1-\bar{X})^2 + \dots + (X_{n}-\bar{X})^2\right)$$
$\sum_{k}(X_{k}-\bar{X})^2=   \sum_{k}X_{k}^2- n\cdot \bar{X}^2    $
$$ \qquad  V=\frac{S_{X}^2}{\frac{\sigma^2}{n-1}}\sim \chi^2_{n-1}  \qquad  $$
\par\noindent - two-tail test for $\sigma$ on level $\alpha$
\vspace*{3mm}
\par $H_0$:  $\sigma=b$ rejected $\quad\Longleftrightarrow\quad$
      $V\leq \chi_{n-1}^2 (\frac{\alpha}2)$ or  $V\geq \chi_{n-1}^2 (1-\frac{\alpha}2)$
\vspace*{5mm}
\par\noindent - two-tailed 95\% confident interval for $\sigma$
    $$\left(\frac{S^2_{X} (n-1)}{\chi^2_{n-1}(0,975)}; \frac{S^2_{X} (n-1)}{\chi^2_{n-1}(0,025)}  \right)$$
